#################################
MiTranscriptome GTF information #
#################################

tcat = transcript category (i.e., lncRNA, TUCP, protein coding, pseudogene, mixed read through)
tstatus = annotation status of the transcript (i.e., does it overlap an annotated gene on the sense strand)
tgenic = indicates whether the given transcript lies between annotated transcripts ("intergenic")
func_name_final = functional name provided to the lncRNA by the MiTranscriptome study
